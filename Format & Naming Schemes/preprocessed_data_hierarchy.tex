			\item \docentry[file]{WcalibTemp.txt} : temporary file containing WEPL calibration curve coefficients.
			\item \docentry[file]{Wcalib.txt} : contains final WEPL calibration curve coefficients.
			\item \docentry[fileset]{$<$Phantom$>$\_XXXX[\_AAA]\_xxx.dat.root.reco.root.bin} : processed experimental data with tracker coordinates, recovery of missing hits when possible, and calibrated WEPL measurements for the object named \docentry![subcategory]{$<$Phantom$>$}, from run \# \docentry![subcategory]{XXXX[\_AAA]}, where \docentry![subcategory]{XXXX} is a 4 digit \# with leading zeros and \docentry![subcategory]{\_AAA} is an optional descriptor tag, and \docentry![subcategory]{xxx} is the gantry angle at which the data was acquired.
	        \end{ThinEnum}
	    \end{ThinEnum}
	\end{ThinEnum}
