\Subsection{Reconstruction Data Hierarchy}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{tcbfunctionenv}'tcbBlueStyle'[title=\shadowText{reconstruction\_data}]
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{tcbparagraph}'tcbBlueStyle'(title=\dirsep ion\dirsep pCT\_data\dirsep reconstruction\_data)%[title=\dirsep ion\dirsep pCT\_data/reconstruction\_data\dirsep$\dots$]
$\boldsymbol{-}$ directory containing the reconstructed images/data.  Although the raw and preprocessed/projection data is not stored in this directory as well, it is organized into the same hierarchy of subdirectories as the organized data to make the data dependencies clear.
\end{tcbparagraph}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{tcbparagraph}'tcbBlueStyle'{tcbenumeratedStyle}
    \begin{deepList}[labelindent=1pt, leftmargin=*]
        \item \coloredtext*{documentation-dir}{\dirsep Phantom} : directory containing all of the experimental/simulated data and reconstructed images associated with this phantom/object.
        \begin{deepList}[labelindent=1pt, leftmargin=*]
            \item \coloredtext*{documentation-constdir}{\dirsep Reference\_Images} : directory containing reference images (xCT, RSP, etc) relevant to analysis/comparison of the data/images for this object and data type.
            \item \coloredtext*{documentation-constdir}{\dirsep Experimental} : directory containing data and images generated from an experimental scan of the object.
            \begin{deepList}[labelindent=1pt, leftmargin=*]
                \item \coloredtext*{documentation-dir}{\dirsep YY-MM-DD} : directory containing data and reconstructed images corresponding to the experimental scan of the object performed on this date.
                \begin{deepList}[labelindent=1pt, leftmargin=*]
                    \item \coloredtext*{documentation-dir}{\dirsep XXXX[\_AAA]} : directory containing data and reconstructed images corresponding to this run \# of the scan date, where the run \# is of the form $\coloredtext*{documentation-subcategory}{``\textit{XXXX}''}$ with optional subcategory tag(s) $\coloredtext*{documentation-subcategory}{``\textit{\_AAA}''}$ specifying additional pertinent information about the scan, such as inferior $\coloredtext*{documentation-subcategory}{``\textit{\_Inf}''}$ or superior $\coloredtext*{documentation-subcategory}{``\textit{\_Sup}''}$ positioning (e.g. for head phantom).
                    \begin{deepList}[labelindent=1pt, leftmargin=*]
                        \item \coloredtext*{documentation-constdir}{\dirsep Input} : directory containing raw data generated by scan of object from each gantry angle and transmitted by event builder.
                        \begin{deepList}[labelindent=1pt, leftmargin=*]
                            \item \coloredtext*{documentation-fileset}{\textit{raw\_xxx.bin}} : binary files containing trigger/tracker/energy detector data from event builder associated with gantry position $\coloredtext*{documentation-subcategory}{``\textit{xxx}''} =\{\coloredtext*{documentation-subcategory}{``\textit{001}''}, \coloredtext*{documentation-subcategory}{``\textit{002}''}, \coloredtext*{documentation-subcategory}{``\textit{003}''}, \cdots\}$.
                        \end{deepList}
                        \item \coloredtext*{documentation-constdir}{\dirsep Output} : directory containing calibration and post processed data generated from analysis of raw data and used as input to image reconstruction.
                        \begin{deepList}[labelindent=1pt, leftmargin=*]
                            \item \coloredtext*{documentation-dir}{\dirsep YY-MM-DD} : directory containing the post processed ``\coloredtext*{documentation-fileset}{\textit{projection\_xxx.bin}}'' data generated on this date and the reconstructions using this data.
                            \begin{deepList}[labelindent=1pt, leftmargin=*]
                                \item \coloredtext*{documentation-file}{\textit{calib.txt}} : text file containing calibration curve coefficients for WEPL calibration.
                                \item \coloredtext*{documentation-file}{\textit{TVcalib.txt}} : text file containing calibration curve coefficients for TV corrected WEPL calibration.
                                \item \coloredtext*{documentation-fileset}{\textit{projection\_xxx.bin}} : preprocessed data files containing tracker coordinates and WEPL data for gantry angle $\coloredtext*{documentation-subcategory}{``\textit{xxx}''} =\{\coloredtext*{documentation-subcategory}{``\textit{001}''}, \coloredtext*{documentation-subcategory}{``\textit{002}''}, \coloredtext*{documentation-subcategory}{``\textit{003}''}, \cdots\}$ used as input to image reconstruction.
						\item \coloredtext*{documentation-constdir}{\dirsep Reconstruction} : directory containing preprocessed data and reconstructed images generated using the ``\coloredtext*{documentation-fileset}{\textit{projection\_xxx.bin}}'' data along with reference images relevant to the object.

                    				\item \coloredtext*{documentation-dir}{\dirsep YY-MM-DD} : directory containing the preprocessed data generated on this date and the reconstructed images generated from this data.
                                    \begin{deepList}[labelindent=1pt, leftmargin=*]

                                        \item \coloredtext*{documentation-file}{\textit{execution\_log.txt}} : execution times for various portions of preprocessing and/or reconstruction and total program execution time.
                                        \item \coloredtext*{documentation-file}{\textit{TV\_measurements.txt}} : text file containing total variation (TV) measurements before/after each iteration
                                        \item \coloredtext*{documentation-file}{\textit{hull.txt}} : text file specifying hull in 1s/0s with the xy plane for each slice stacked on each other.
                                        \item \coloredtext*{documentation-file}{\textit{hull.png}} : ``\coloredtext*{documentation-file}{\textit{hull.txt}}'' converted to PNG image.
