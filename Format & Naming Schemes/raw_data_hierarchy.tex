\Subsection*{Raw Data Hierarchy}
\begin{tcbenvironment}'tcbBlueStyle'[title=\shadowText{raw\_data}]
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{tcbparagraph}'tcbBlueStyle'(title=\dirsep ion\dirsep pCT\_data\dirsep raw\_data)%[title=\dirsep ion\dirsep pCT\_data\dirsep raw\_data\dirsep$\dots$]
$\boldsymbol{-}$ directory where all raw experimental data files from a particular scan are stored in separate directories according to the scan date prior to creation of soft symbolic links named \docentry![fileset]{projection\_xxx.bin} and organized according to the naming/organizational scheme.
\end{tcbparagraph}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{tcbparagraph}'tcbBlueStyle'{tcbenumeratedStyle}
\begin{ThinEnum}[labelindent=1pt, leftmargin=*]
    \item \docentry[dir]{YY-MM-DD} : Folder containing all raw experimental data acquired from the scan beginning on \docentry![subcategory]{YY-MM-DD}
    \begin{ThinEnum}[labelindent=1pt, leftmargin=*]
        \item \docentry[fileset]{$<$Phantom$>$\_XXXX[\_AAA]\_xxx.dat} : raw experimental data for the object named \docentry![subcategory]{$<$Phantom$>$}, from run \# \docentry![subcategory]{XXXX[\_AAA]}, where \docentry![subcategory]{XXXX} is a 4 digit \# with leading zeros and \docentry![subcategory]{\_AAA} is an optional descriptor tag, and \docentry![subcategory]{xxx} is the gantry angle at which the data was acquired.
    \end{ThinEnum}
\end{ThinEnum}
\end{tcbparagraph}
\end{tcbenvironment}
