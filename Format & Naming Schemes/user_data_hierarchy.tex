\Subsection{Unorganized User Data Hierarchy}
\begin{tcbenvironment}'tcbBlueStyle'[title=\shadowText{user\_data}]

\begin{tcbparagraph}'tcbBlueStyle'(title=\dirsep ion\dirsep pCT\_data\dirsep user\_data)%[title=\dirsep local\dirsep user\_data\dirsep$\dots$]
$\boldsymbol{-}$ directory containing unorganized input and output reconstruction data, allowing users to use and keep their unorganized data separate from other data and maintain it in their preferred organizational scheme without it interfering with the properly organized data.
\end{tcbparagraph}

    \begin{ThinEnum}[labelindent=1pt, leftmargin=*]
		\item \docentry[dir]{\caratenclosed*{username}} : subdirectories for each pCT user for the unorganized input and output reconstruction data
	\end{ThinEnum}
\end{ThinEnum}
\end{tcbparagraph}
