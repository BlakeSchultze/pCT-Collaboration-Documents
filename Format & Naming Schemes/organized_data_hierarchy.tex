\Subsection{Organized Data Hierarchy}

\begin{tcbfunctionenv}'tcbBlueStyle'[title=\shadowText{organized\_data}]%, bottom=25.0mm
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{tcbparagraph}'tcbBlueStyle'(title=\dirsep ion\dirsep pCT\_data\dirsep organized\_data)%[title=\dirsep ion\dirsep pCT\_data\dirsep organized\_data\dirsep$\dots$]
$\boldsymbol{-}$ directory containing all of the pCT data (raw, preprocessed/projection, and reconstruction), primarily as soft links to the actual data itself (with each type of data stored/organized in separate directories), organized in a hierarchy of subdirectories indicating data dependencies.
\end{tcbparagraph}

\item \coloredtext*{documentation-file}{\textit{bin\_counts.txt}} : linearized bin \# for each proton history, where linearized bin \# = t\_bin + angle\_bin * T\_BINS + v\_bin * T\_BINS * ANGULAR\_BINS.
\item {\coloredtext*{documentation-file}{\textit{mean\_rel\_ut\_angle.txt}} : mean relative ut angle ($\angle ut_{out} -\angle ut_{in}$) by linearized bin \#.}
\item \coloredtext*{documentation-file}{\textit{mean\_rel\_uv\_angle.txt}} : mean relative uv angle ($\angle uv_{out} -\angle uv_{in}$) by linearized bin \#.
\item \coloredtext*{documentation-file}{\textit{mean\_WEPL.txt}} : mean WEPL value by linearized bin \#.
\item \coloredtext*{documentation-file}{\textit{stddev\_rel\_ut\_angle.txt}} : standard deviation of the relative ut angle ($\angle ut_{out} -\angle ut_{in}$) by linearized bin \#.
\item \coloredtext*{documentation-file}{\textit{stddev\_rel\_uv\_angle.txt}} : standard deviation of the relative uv angle ($\angle uv_{out} -\angle uv_{in}$) by linearized bin \#.
\item \coloredtext*{documentation-file}{\textit{stddev\_WEPL.txt}} : standard deviation of the WEPL value by linearized bin \#.
\item \coloredtext*{documentation-file}{\textit{sinogram.txt}} : mean WEPL after statistical cuts with the $t_{bin}$ and angular bin $\theta_{bin}$ plane for each vertical bin $v_{bin}$ stacked on each other.
\item \coloredtext*{documentation-file}{\textit{sin\_table.bin}} : file containing the tabulated values of sine function
\item \coloredtext*{documentation-file}{\textit{cos\_table.bin}} : file containing the tabulated values of cosine function
\item \coloredtext*{documentation-file}{\textit{coefficient.bin}} : file containing the tabulated scattering coefficient values for $\Sigma_1$/$\Sigma_2$ for $u_2-u_1$/$u_1$ values
\item \coloredtext*{documentation-file}{\textit{poly\_1\_2.bin}} : file containing the tabulated MLP polynomial values with coefficients $\{1,2,3,4,5,6\}$
\item \coloredtext*{documentation-file}{\textit{poly\_2\_3.bin}} : file containing the tabulated MLP polynomial values with coefficients $\{2,3,4,5,6,7\}$
\item \coloredtext*{documentation-file}{\textit{poly\_3\_4.bin}} : file containing the tabulated MLP polynomial values with coefficients $\{3,4,5,6,7,8\}$
\item \coloredtext*{documentation-file}{\textit{poly\_2\_6.bin}} : file containing the tabulated MLP polynomial values with coefficients $\{2,6,12,20,30,42\}$
\item \coloredtext*{documentation-file}{\textit{poly\_3\_12.bin}} : file containing the tabulated MLP polynomial values with coefficients $\{3,12,30,60,105,168\}$
\item \coloredtext*{documentation-file}{\textit{MLP.bin}} : binary file with MLP path data for each history entering hull.
\item \coloredtext*{documentation-file}{\textit{WEPL.bin}} : binary file specifying WEPL value for each history entering hull.
\item \coloredtext*{documentation-file}{\textit{histories.bin}} : binary file specifying entry/exit coordinates/angles, bin number, gantry angle, and hull entry x/y/z voxel \# for each history entering hull.
